3D Pie chart

This is an example of a 3D Pie chart effect then can be accomplished with a few extra lines of code.

[No canvas support]

This goes in the documents header (or you could place it just above the jQuery ready event code): 
<script src="/mps/libs/jquery-1.11.3/jquery-1.11.3.js"></script>
<script src="RGraph.common.core.js"></script>
<script src="RGraph.common.dynamic.js"></script>
<script src="RGraph.common.tooltips.js"></script>
<script src="RGraph.common.effects.js"></script>
<script src="RGraph.pie.js"></script>
Put this where you want the chart to show up: 
<canvas id="cvs" width="450" height="300">
    [No canvas support]
</canvas>
This is the code that generates the chart. Because it's using the jQuery ready event you can put this at the bottom of the document: 
<script>
    $(document).ready(function ()
    {
        var pie = new RGraph.Pie({
            id: 'cvs',
            data: [4,5,5,7,5,8,4,9],
            options: {
                strokestyle: '#e8e8e8',
                linewidth: 2,
                shadowOffsetx: 0,
                shadowOffsety: 15,
                shadowColor: '#aaa',
                exploded: 10,
                radius: 80,
                tooltips: ['Mavis','Kevin','Luis','June','Olga','Luis','Pete','Bridget'],
                tooltipsCoordsPage: true,
                labels: ['Mavis','Kevin','Luis','June','Olga','Luis','Pete','Bridget'],
                labelsSticks: true,
                labelsSticksLength: 15
            }
        })
        
        // This is the factor that the canvas is scaled by
        var factor = 1.5;
    
        // Set the transformation of the canvas - a scale up by the factor (which is 1.5 and a simultaneous translate
        // so that the Pie appears in the center of the canvas
        pie.context.setTransform(factor,0,0,1,((pie.canvas.width * factor) - pie.canvas.width) * -0.5,0);
    
        //pie.draw();
        pie.roundRobin({frames: 30})
    });
</script>
