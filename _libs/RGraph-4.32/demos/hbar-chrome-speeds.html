A horizontal bar chart showing Chrome speeds

[No canvas support]

This goes in the documents header (or you could place it just above the jQuery ready event code): 
<script src="/mps/libs/jquery-1.11.3/jquery-1.11.3.js"></script>
<script src="RGraph.common.core.js"></script>
<script src="RGraph.hbar.js"></script>
Put this where you want the chart to show up: 
<canvas id="cvs" width="350" height="176">
    [No canvas support]
</canvas>
This is the code that generates the chart. Because it's using the jQuery ready event you can put this at the bottom of the document: 
<script>
    $(document).ready(function ()
    {
        var hbar = new RGraph.HBar({
            id: 'cvs',
            data: [215, 175],
            options: {
                labels: ['Chrome 18', 'Chrome 19'],
                colors: ['#3266CC'],
                title: 'SunSpider score',
                titleXaxis: 'milliseconds (less is better)',
                titleXaxisY: 155,
                backgroundGridHlines: false,
                backgroundGridBorder: false,
                backgroundGridAutofitNumvlines: 4,
                gutterBottom: 55,
                vmargin: 8,
                gutterTop: 35,
                gutterLeft: 85
            }
        }).draw();
   });
</script>
